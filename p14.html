C program to find the greatest number among three numbers. All three numbers have to be taken as an input from the user.

#include
int main()
{
int n1, n2, n3;
printf("n1: \n");
scanf("%d", &n1);
printf("n2: \n");
scanf("%d", &n2);
printf("n3: \n");
scanf("%d", &n3);
if (n1 > n2 && n1 > n3)
{
printf("n1 is big");
}
else if (n2 > n1 && n2 > n3)
{
printf("n2 is big");
}
else
{
printf("n3 is big");
}
return 0;
}
