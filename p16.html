Write a C program to input week day number(1-7) and print the corresponding day of week name using if else.

#include
int main()
{
int day;
printf("enter day number(1-7)");
scanf("%d", &day);
if(day==1)
printf("it is monday");
else if(day==2)
printf("it is tuesday");
else if(day==3)
printf("it is wednsday");
else if(day==4)
printf("it is thursday");
else if(day==5)
printf("it is friday");
else if(day==6)
printf("it is saturday");
else if(day==7)
printf("it is sunday");
else
printf("not valid");
return 0;
}
