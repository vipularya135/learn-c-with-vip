Write a program to implement a calculator using functions. Use call by value.

#include
int choice(int ch, int a,int b);
int main()
{
int a,b,ch;
printf("Enter the numbers \n");
scanf("%d %d", &a, &b);
printf("Enter 1 for Addition, 2 for Subtraction, 3 for Multiplication, 4 for Division \n");
scanf( "%d", &ch );
choice(ch, a, b);
}
int choice(int ch,int a, int b)
{
int sum,sub,mul,div;
if(ch==1)
{
sum=a+b;
printf("The sum is %d \n", sum);
}
else if(ch==2)
{
sub=a-b;
printf("The difference is %d \n", sub);
}
else if(ch==3)
{
mul=a*b;
printf("The product is %d \n", mul);
}
else if(ch==4)
{
div=a/b;
printf( "The quotient is %d \n", div);
}
else
printf("Invalid choice \n");
}
