Write a program to store and print the roll no., name , age and semester of n student using structures.

#include
int main()
{

int roll;
char name;
int age;

} s[n];
printf("enter number:\n");
scanf("%d", &n);

for(int i=1;i<=n;i++)
{
scanf(" %d %c %d %d", &s[i].roll, &s[i].name, &s[i].age, &s[i].sem);
