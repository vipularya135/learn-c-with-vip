Write a program to enter details of two students(name, age, roll no.) using input function. Pass the details to the output function and print the details of the students. Use structure for storing student information.

#include
struct stu{
int roll;
int age;
char name[100];
}b;
int input();
int output();
int main()
{
input();
output();
}
int input()
{
printf("Enter the name of the student: ");
scanf("%s", b.name);
printf("Enter the age of the student: ");
scanf("%d", &b.age);
printf("Enter the roll number of the student: ");
scanf("%d", &b.roll);
return 0;
}
int output(struct stu *ptr)
{
printf("Details of the student are: \n");
printf("Name: %s\n", ptr->name);
printf("Age: %d\n", ptr->age);
printf("Roll number: %d\n", ptr->roll);
return 0;
}
