Enter the marks of 'n' students in Chemistry, Mathematics and Physics (each out of 100) using a structure named Marks having elements roll no., name, chem_marks, maths_marks and phy_marks and then display the percentage of each student.

#include
#include
int main()
{
int n, i;
struct Marks {
char name[20];
int perc;
int rno, chem_marks, math_marks, phy_marks;
};
struct Marks a[n];
printf("Enter the number of students \n");
scanf("%d", &n);
int sum;
printf("Enter name,roll no., and marks out of 100.\n");
for (i = 0; i < n; i++)
{
scanf(" %s %d %d %d %d", &a[i].name, &a[i].rno, &a[i].chem_marks,
&a[i].math_marks, &a[i].phy_marks);
a[i].perc = (a[i].chem_marks + a[i].math_marks + a[i].phy_marks) / 3;
printf("%d.%s: %d%\n", a[i].rno, a[i].name, a[i].perc);
}
}
