Write a structure to store the names, salary and hours of work per day of 10 employees in a company. Write a program to increase the salary depending on the number of hours of work per day as follows and then print the name of all the employees along with their final salaries.
if 8hrs - increase 50, 10 - 100, >12 - 150

#include
#include
int main()
{
struct emp {
char name[100];
int hrs, salary;
};
int n=3;
struct emp a[n];
printf("Enter name,salary,hours \n");
for (int i = 0; i < n; i++)
{
scanf(" %s %d %d", &a[i].name, &a[i].salary, &a[i].hrs);
if (a[i].hrs == 8)
{
a[i].salary += 50;
}
else if (a[i].hrs == 10)
{
a[i].salary += 100;
}
else if (a[i].hrs >= 12)
{
a[i].salary += 150;
}
}
printf("Name\tSalary \tHours Worked \n");
for (int i = 0; i < n; i++)
{
printf("%s\t%d \t%d \n", a[i].name, a[i].salary, a[i].hrs);
}
}
