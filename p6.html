Write a program to take marks of 5 subjects as input from the keyboard. Find the percentage. If percentage is Greater than 60 -> First Division If percentage is Between 50 and 60 -> Second Division If percentage is Between 40 and 50 -> Third Division If percentage is Less than 40 -> Fail.

#include int main()
{
int a,b,c,d,e,f;
printf("first subject: \n");
scanf("%d",&a);
printf("second subject: \n");
scanf("%d",&b);
printf("third subject: \n");
scanf("%d",&c);
printf("fourth subject: \n");
scanf("%d",&d);
printf("fifth subject: \n");
scanf("%d",&e);
f=(a+b+c+d+e)/5;
printf("%d",f);
if(f>60&&f<=100)
{
printf("first division");
}
else if(f>50&&f<=60)
{
printf("second division");
}
else if(f>40&&f<=50)
{
printf("third division");
}
else
{
printf("FAIL");
}
return 0;
}
